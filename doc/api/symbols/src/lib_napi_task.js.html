1 NApi._.Linfo("NApi.task");
  2 
  3 /** @namespace NApi.task
  4  * Параметры задания
  5  */
  6 NApi.task = {
  7 	/** @namespace NApi.task._
  8 	 * Функционал, используемый только внутри модуля NApi.task
  9 	 * @private
 10 	 */
 11 	_ : {
 12 		current_task : undefined
 13 	},
 14 
 15 	/** @class NApi.task.Task
 16 	 * Задание
 17 	 * @property {String} text Текст задания
 18 	 * @property {String} analys Текст разбора задания
 19 	 * @property {Array} answers Правильные ответы
 20 	 * @property {Array} wrong_answers Неправильные ответы
 21 	 * @property {Array} tags Теги
 22 	 * @property {Function} check_answer Функция проверки ответа
 23 	 * @property {Function} draw Функция отрисовки
 24 	 *
 25 	 * @constructor NApi.task.Task
 26 	 * @param {String} text текст задания
 27 	 * @param {String} analys текс разбора задания
 28 	 * @param {Array} answers правильные ответы
 29 	 * @param {Array} wrong_answers неправильные ответы
 30 	 * @param {Array} tags теги
 31 	 * @param {Function} check_answer функция проверки ответа
 32 	 * @param {Function} draw функция отрисовки
 33 	 */
 34 	Task : function(text, analys, answers, wrong_answers, tags, check_answer, draw) {
 35 		return {
 36 			/**
 37 			 * @default ""
 38 			 */
 39 			text : text || "",
 40 			analys : analys || "",
 41 			answers : answers || [],
 42 			wrong_answers : wrong_answers || [],
 43 			tags : tags || [],
 44 			check_answer : check_answer || undefined,
 45 			draw : draw || undefined
 46 		}
 47 	},
 48 
 49 	/** @function NApi.task.set_current_task
 50 	 * Установить текущее задание
 51 	 * @param {NApi.task.Task} task задание
 52 	 */
 53 	set_current_task : function(task) {
 54 		NApi.task._.current_task = task;
 55 
 56 		window.vopr.podg();
 57 
 58 		if (task != undefined) {
 59 			window.vopr.txt = task.text;
 60 			window.vopr.rsh = task.analys;
 61 			window.vopr.ver = task.answers;
 62 			window.vopr.nev = task.wrong_answers;
 63 			if (task.check_answer != undefined) {
 64 				window.vopr.vrn = task.check_answer;
 65 			}
 66 			if (task.draw != undefined) {
 67 				window.vopr.dey = task.draw;
 68 			}
 69 
 70 			for (var t in task.tags) {
 71 				window.vopr[task.tags[t]] = 0;
 72 			}
 73 		}
 74 	},
 75 
 76 	/** @function NApi.task.get_current_task
 77 	 * @return текущее задание
 78 	 */
 79 	get_current_task : function() {
 80 		return NApi.task._.current_task;
 81 	}
 82 }
