1 /** @namespace NApi.task
  2  * Параметры задания
  3  */
  4 NApi.task = {
  5 	/** @namespace NApi.task._
  6 	 * Функционал, используемый только внутри модуля NApi.task
  7 	 * @private
  8 	 */
  9 	_ : {},
 10 
 11 	/** @class NApi.task.Task
 12 	 * Задание
 13 	 * @property {String} text Текст задания
 14 	 * @property {String} analys Текст разбора задания
 15 	 * @property {Array} answers Правильные ответы
 16 	 * @property {Array} wrong_answers Неправильные ответы
 17 	 * @property {Array} tags Теги
 18 	 * @property {Function} check_answer Функция проверки ответа
 19 	 * @property {Function} draw Функция отрисовки
 20 	 *
 21 	 * @constructor NApi.task.Task
 22 	 * @param {String} text текст задания
 23 	 * @param {String} analys текс разбора задания
 24 	 * @param {Array} answers правильные ответы
 25 	 * @param {Array} wrong_answers неправильные ответы
 26 	 * @param {Array} tags теги
 27 	 * @param {Function} check_answer функция проверки ответа
 28 	 * @param {Function} draw функция отрисовки
 29 	 */
 30 	Task : function(text, analys, answers, wrong_answers, tags, check_answer, draw) {
 31 		return {
 32 			text : text || "",
 33 			analys : analys || "",
 34 			answers : answers || [],
 35 			wrong_answers : wrong_answers || [],
 36 			tags : tags || [],
 37 			check_answer : check_answer || undefined,
 38 			draw : draw || undefined
 39 		}
 40 	},
 41 
 42 	/** @function NApi.task.set_current_task
 43 	 * Установить задание
 44 	 * @param {String} text текст задания
 45 	 * @param {String} analys текс разбора задания
 46 	 * @param {Array} answers правильные ответы
 47 	 * @param {Array} wrong_answers неправильные ответы
 48 	 * @param {Array} tags теги
 49 	 * @param {Function} check_answer функция проверки ответа
 50 	 * @param {Function} draw функция отрисовки
 51 	 */
 52 	set_task : function(text, analys, answers, wrong_answers, tags, check_answer, draw) {
 53 		window.vopr.podg();
 54 
 55 		window.vopr.txt = text;
 56 		window.vopr.rsh = analys;
 57 		window.vopr.ver = answers;
 58 		window.vopr.nev = wrong_answers;
 59 		if (check_answer) {
 60 			window.vopr.vrn = check_answer;
 61 		}
 62 		if (draw) {
 63 			window.vopr.dey = draw;
 64 		}
 65 
 66 		if (tags) {
 67 			for (var t in tags) {
 68 				window.vopr.kat[tags[t]] = 0;
 69 			}
 70 		}
 71 	}
 72 }
