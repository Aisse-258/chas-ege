  3 /** @namespace NApi.task
  4  * Параметры задания

  6 NApi.task = {
  7 	/** @namespace NApi.task._
  8 	 * Функционал, используемый только внутри модуля NApi.task
  9 	 * @private
 10 	 */
 11 	_ : {},
 12 
 13 
 14 	/** @function NApi.task.setTask
 15 	 * Установить задание
 16 	 * @param {String} text текст задания
 17 	 * @param {String} analys текс разбора задания
 18 	 * @param {Array} answers правильные ответы
 19 	 * @param {Array} wrongAnswers неправильные ответы
 20 	 * @param {Array} tags теги
 21 	 * @param {Function} checkAnswer функция проверки ответа
 22 	 * @param {Function} draw функция отрисовки
 23 	 */
 24 	setTask : function(o) {
 25 		window.vopr.podg();
 26 
 27 		window.vopr.txt = o.text || "";
 28 		window.vopr.rsh = o.analys || "";
 29 		window.vopr.ver = o.answers || [];
 30 		window.vopr.nev = o.wrongAnswers || [];
 31 		if (o.checkAnswer) {
 32 			window.vopr.vrn = checkAnswer;
 33 		}
 34 		if (o.draw) {
 35 			window.vopr.dey = draw;
 36 		}
 37 
 38 		if (o.tags) {
 39 			for (var t in o.tags) {
 40 				window.vopr.kat[o.tags[t]] = 0;
 41 			}
 42 		}
 43 	}
 44 }
