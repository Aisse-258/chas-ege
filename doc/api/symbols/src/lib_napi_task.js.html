1 "use strict";
  2 
  3 /** @namespace NApi.task
  4  * Параметры задания
  5  */
  6 NApi.task = {
  7 	/** @namespace NApi.task._
  8 	 * Функционал, используемый только внутри модуля NApi.task
  9 	 * @private
 10 	 */
 11 	_ : {},
 12 
 13 
 14 	/** @function NApi.task.setTask
 15 	 * Установить задание
 16 	 * @param {String} text текст задания
 17 	 * @param {String} analys текс разбора задания
 18 	 * @param {String|Number|String[]|Number[]} answers правильные ответы
 19 	 * @param {String|Number|String[]|Number[]} wrongAnswers неправильные ответы
 20 	 * @param {String[]} tags теги
 21 	 * @param {Function} checkAnswer функция проверки ответа
 22 	 * @param {Function} draw функция отрисовки
 23 	 */
 24 	setTask : function(o) {
 25 		window.vopr.podg();
 26 
 27 		var text = o.text || "";
 28 		var analys = o.analys || "";
 29 		var answers = o.answers || [];
 30 		var wrongAnswers = o.wrongAnswers || [];
 31 		var checkAnswer = o.checkAnswer;
 32 		var draw = o.draw;
 33 		var tags = o.tags;
 34 
 35 		if (NLib.getTypeOf(text) != "[object String]") {
 36 			throw TypeError("Параметр text должен быть строкой");
 37 		}
 38 
 39 		if (NLib.getTypeOf(analys) != "[object String]") {
 40 			throw TypeError("Параметр analys должен быть строкой");
 41 		}
 42 
 43 		if (checkAnswer != undefined && NLib.getTypeOf(checkAnswer) != "[object Function]") {
 44 			throw TypeError("Параметр checkAnswer должен быть функцией или отсутствовать");
 45 		}
 46 
 47 		if (draw != undefined && NLib.getTypeOf(draw) != "[object Function]") {
 48 			throw TypeError("Параметр draw должен быть функцией или отсутствовать");
 49 		}
 50 
 51 		if (tags != undefined && NLib.getTypeOf(tags) != "[object Array]") {
 52 			throw TypeError("Параметр tags должен быть массивом строк или отсутствовать");
 53 		}
 54 		for (var t in tags) {
 55 			if (NLib.getTypeOf(t) != "[object String]") {
 56 				throw TypeError("Параметр tags (массив) должен содержать только строки");
 57 			}
 58 		}
 59 
 60 		answers = NLib.toStringsArray(answers);
 61 		wrongAnswers = NLib.toStringsArray(wrongAnswers);
 62 
 63 		window.vopr.txt = text;
 64 		window.vopr.rsh = analys;
 65 		window.vopr.ver = answers;
 66 		window.vopr.nev = wrongAnswers;
 67 		if (checkAnswer) {
 68 			window.vopr.vrn = checkAnswer;
 69 		}
 70 		if (draw) {
 71 			window.vopr.dey = draw;
 72 		}
 73 
 74 		if (tags) {
 75 			for (var t in tags) {
 76 				window.vopr.kat[tags[t]] = 0;
 77 			}
 78 		}
 79 
 80 		var voprcheck = dvig.validateVopr();
 81 		if (voprcheck) {
 82 			NApi._.Lwarn("Результат проверки вопроса:\n\t" + voprcheck);
 83 		}
 84 	}
 85 }
