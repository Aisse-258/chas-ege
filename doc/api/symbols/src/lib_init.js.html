116 function checkJQuery(fun,pole){
117 	try{
118 		!jQuery;//Вызовет ошибку, если jQuery не загружена
119 		return 1;
120 	}catch(e){
121 		zagr(chas.libs.localJQuery+chas.libs.fileJQuery+'?'+Math.random());
122 		if(pole)
123 			document.getElementById(pole).innerHTML='Проблемы с внешней библиотекой JQuery. Устранение неполадок, подождите...';
124 		if(fun)
125 			setTimeout(fun,1000);
126 		return 0;
127 	}
128 }

130 function checkMathJax(fun,pole){
131 	try{
132 		!MathJax;//Вызовет ошибку, если MathJax не загружен
133 		return 1;
134 	}catch(e){
135 		zagr(chas.libs.remoteMathJax+chas.libs.fileMathJax);
136 		zagr(chas.libs. localMathJax+chas.libs.fileMathJax);
137 		if(pole)
138 			$('#'+pole).html('Проблемы при обращении к библиотеке отрисовки формул, пытаемся дотянуться...');
139 		if(fun)
140 			setTimeout(fun,1000);
141 		return 0;
142 	}
143 }
