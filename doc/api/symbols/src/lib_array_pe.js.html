  3 Array.prototype.pe_inv=function(){
  4 /**Количество инверсий в перестановке, образованной элементами массива.*/
  5 	var perest=0;
  6 	var len=this.length;
  7 	for(var i=0;i<len;i++)
  8 		for(var j=i;j<len;j++)
  9 			if(this[i]>this[j])
 10 				perest++;
 11 	return perest;
 12 }
