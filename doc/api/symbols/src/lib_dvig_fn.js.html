 18 function fn_promezh(a){
 19 	if(a.prnz)
 20 		a.prnz=a.prnz.ts().plusminus();
 21 	if(a.prkz)
 22 		a.prkz=a.prkz.ts().plusminus();
 23 	//Если границ промежутка нет, то это числовая прямая.
 24 	if(a.prnz==undefined && a.prkz==undefined)
 25 		return {
 26 			nazv:'',
 27 			text:'$(-\\infty;\\infty)$',
 28 		}
 29 	
 30 	if(a.prnz==undefined && a.prkb)
 31 		return{
 32 			nazv:'открытый луч',
 33 			text:'$(-\\infty;'+a.prkz.ts()+')$',
 34 		}
 35 	
 36 	if(a.prkz==undefined && a.prnb)
 37 		return{
 38 			nazv:'открытый луч',
 39 			text:'$('+a.prnz.ts()+';\\infty)$',
 40 		}
 41 	
 42 	if(a.prnz==undefined)
 43 		return{
 44 			nazv:'луч',
 45 			text:'$(-\\infty;'+a.prkz.ts()+']$',
 46 		}
 47 	
 48 	if(a.prkz==undefined)
 49 		return{
 50 			nazv:'луч',
 51 			text:'$['+a.prnz.ts()+';\\infty)$',
 52 		}
 53 
 54 	//Ага, обе границы есть
 55 	
 56 	if(a.prkb&&a.prnb)
 57 		return{
 58 			nazv:'интервал',
 59 			text:'$('+a.prnz.ts()+';'+a.prkz.ts()+')$',
 60 		}
 61 	
 62 	if(a.prkb)
 63 		return{
 64 			nazv:'полуинтервал',
 65 			text:'$['+a.prnz.ts()+';'+a.prkz.ts()+')$',
 66 		}
 67 	
 68 	if(a.prnb)
 69 		return{
 70 			nazv:'полуинтервал',
 71 			text:'$('+a.prnz.ts()+';'+a.prkz.ts()+']$',
 72 		}
 73 	
 74 	return{
 75 		nazv:'отрезок',
 76 		text:'$['+a.prnz.ts()+';'+a.prkz.ts()+']$',
 77 	}
 78 }
