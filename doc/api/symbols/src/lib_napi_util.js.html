 12 	/** @function NApi.util.math.get_random_items
 13 	 * Получение случайные (не повторяющиеся) элементы массива
 14 	 * @param {Array} arr массив
 15 	 * @param {Number} count кол-во случайных элеметнов
 16 	 * @return Массив случайных элементов массива
 17 	 */
 18 	get_random_items : function(arr, count) {
 19 		var items = [];
 20 		for (var i = 0; i < count; i++) {
 21 			if (arr.length >= 1) {
 22 				items.push(arr[NApi.util.math.randomize(0, arr.length - 1)]);
 23 			} else {
 24 				items.push(arr[0]);
 25 			}
 26 		}
 27 		return items;
 28 	},

 68 		/** @function NApi.util.math.round
 69 		 * Округление числа
 70 		 * @param {Number} num округляемое число
 71 		 * @param {Number} dec_places точность округления
 72 		 * @return число num округлённое до dp
 73 		 */
 74 		round : function(num, dec_places) {
 75 			var dec_places = dec_places || 1;
 76 			return Math.round(num / dec_places) * dec_places;
 77 		},

 80 		/** @function NApi.util.math.randomize
 81 		 * Генерация случайного числа
 82 		 * @param {Number} min минимальное значение
 83 		 * @param {Number} max минимальное значение
 84 		 * @param {Number} dec_places кол-во знаков после запятой
 85 		 * @return случайное число
 86 		 */
 87 		randomize : function(min, max, dec_places) {
 88 			return NApi.util.math.round(Math.random() * ((max || 9007199254740992) - (min || 0)), (dec_places || 1));
 89 		},

 92 		/** @function NApi.util.math.is_divisor
 93 		 * Проверка кратности
 94 		 * @param {Number} num делимое
 95 		 * @param {Number} divisor делитель
 96 		 * @return Кратно ли число a числу divisor
 97 		 */
 98 		is_divisor : function(num, divisor) {
 99 			return num % divisor == 0;
100 		},

103 		/** @function NApi.util.math.get_divisors
104 		 * Получение делителей числа
105 		 * @param {Number} num делимое
106 		 * @return Массив делителей числа num
107 		 */
108 		get_divisors : function(num) {
109 			if (num == 0) {
110 				return [1];
111 			}
112 			var divisors = [];
113 			for (var d = 1; d <= num; d++) {
114 				if (NApi.util.math.is_divisor(num, d)) {
115 					divisors.push(d);
116 				}
117 			}
118 			return divisors;
119 		},

134 /** @function NApi.util.math.Triangle.get_area
135  * Вычисление площади
136  * @return площадь треугольника
137  * @memberof NApi.util.math.Triangle#
138  */
139 NApi.util.math.Triangle.prototype.get_area = function() {
140 	return 0.5 * ((this.a.x * this.b.y) + (this.a.y * this.c.x) + (this.b.x * this.c.y) - (this.b.y * this.c.x) -
141 	              (this.c.y * this.a.x) - (this.a.y - this.b.x)).abs();
142 }
