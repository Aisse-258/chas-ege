 12 	/** @function NApi.util.round
 13 	 * Округление числа
 14 	 * @param {Number} num округляемое число
 15 	 * @param {Number} dec_places точность округления
 16 	 * @return число num округлённое до dp
 17 	 */
 18 	round : function(num, dec_places) {
 19 		return Math.round(num / dec_places) * dec_places;
 20 	},

 23 	/** @function NApi.util.randomize
 24 	 * Генерация случайного числа
 25 	 * @param {Number} min минимальное значение
 26 	 * @param {Number} max минимальное значение
 27 	 * @param {Number} dec_places кол-во знаков после запятой
 28 	 * @return случайное число
 29 	 */
 30 	randomize : function(min, max, dec_places) {
 31 		return NApi.util.round(Math.random() * (max - min), (dec_places || 1));
 32 	}
