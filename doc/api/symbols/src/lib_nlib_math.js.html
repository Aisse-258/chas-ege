 16 	/** @function NLib.math.randomize
 17 	 * Генерация случайного числа
 18 	 * @param {Number} min минимальное значение
 19 	 * @param {Number} max минимальное значение
 20 	 * @param {Number} dec_places кол-во знаков после запятой
 21 	 * @return случайное число
 22 	 */
 23 	randomize : function(min, max, dec_places) {
 24 		return (Math.random() * ((max || 9007199254740992) - (min || 0))).round(dec_places || 1);
 25 	}
