  3 function Complex(re,im){
  4 	this.re=+re?+re:0;
  5 	this.im=+im?+im:0;
  6 	this.isComplex=1;
  7 	this.toString=function(){
  8 	/**Представляет число в виде a+bi*/
  9 		var rez;
 10 		if(!this.re && !this.im)
 11 			rez='0';
 12 		else if( this.re && !this.im)
 13 			rez=''+this.re;
 14 		else if(!this.re &&  this.im)
 15 			rez=''+this.im+'i';
 16 		else if( this.re &&  this.im)
 17 			rez= ''+this.re+'+'+this.im+'i';
 18 		return rez.plusminus();
 19 	}
 20 	this.ts=function(){
 21 		return this.toString().ts();
 22 	}
 23 
 24 	this.minus=function(){
 25 	/**Противоположное число: -a-bi*/
 26 		return new Complex( -(this.re), - (this.im));
 27 	}
 28 
 29 	this.sopr=function(){
 30 	/**Сопряжёное число: a-bi*/
 31 		return new Complex(this.re, - (this.im));
 32 	}
 33 
 34 	this.abs=
 35 	this.norma=function(){
 36 	/**Норма (модуль, абсолютное значение) комплексного числа*/
 37 		return (this.re.sqr()+this.im.sqr()).sqrt();
 38 	}
 39 
 40 	this.obrat=function(){
 41 	/**Обратное число: a-bi*/
 42 		var n=this.norma().sqr();
 43 		return new Complex(this.re/n, - (this.im)/n);
 44 	}
 45 	
 46 	this.sum=function(){
 47 	/**Прибавить к комплексному числу комплексные или действительные*/
 48 		var rez=this.clone();
 49 //		arguments[0].isComplex?arguments[0].clone():new Complex(arguments[0]);
 50 		for(var i=arguments.length-1;i>=0;i--){
 51 			var operand=arguments[i];
 52 			if(operand.isNumber){
 53 				rez.re+=operand;
 54 			}else{
 55 				if(operand.re){
 56 					rez.re+=operand.re;
 57 				}
 58 				if(operand.im){
 59 					rez.im+=operand.im;
 60 				}
 61 			}
 62 		}
 63 		return rez;
 64 	}
 65 
 66 	this.umn=function(){
 67 	/**Умножить комплексное число на комплексные или действительные*/
 68 		var rez=this.clone();
 69 		for(var i=arguments.length-1;i>=0;i--){
 70 			var operand=arguments[i];
 71 			if(operand.isNumber){
 72 				if(operand===0){
 73 					return new Complex();
 74 				}
 75 				rez.re*=operand;
 76 				rez.im*=operand;
 77 			}else{
 78 				if(!operand.re && !operand.im){
 79 					//Нуль
 80 					return new Complex();
 81 				}else{
 82 					var r=rez.re,
 83 						m=rez.im;
 84 					rez.re=r*operand.re-m*operand.im;
 85 					rez.im=r*operand.im+m*operand.re;
 86 				} 
 87 			}
 88 		}
 89 		return rez;
 90 	}
