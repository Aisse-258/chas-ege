1 "use strict";
  2 
  3 /** @namespace NApi.debug
  4  * Параметры задания
  5  */
  6 NApi.debug = {
  7 	/** @namespace NApi.debug._
  8 	 * Функционал, используемый только внутри модуля NApi.debug
  9 	 * @private
 10 	 */
 11 	_ : {
 12 		debug_mode : false,
 13 	},
 14 
 15 
 16 	isDebug : function() {
 17 		return NApi.debug._.debug_mode;
 18 	},
 19 
 20 
 21 	/** @function NApi.debug.enableDebugging
 22 	 * Включить режим отладки
 23 	 */
 24 	enableDebugging : function() {
 25 		NApi.debug._.debug_mode = true;
 26 	},
 27 
 28 
 29 	/** @function NApi.debug.disableDebugging
 30 	 * Выключить режим отладки
 31 	 */
 32 	disableDebugging : function() {
 33 		NApi.debug._.debug_mode = false;
 34 	},
 35 
 36 
 37 	/** @function NApi.debug.Ldebug
 38 	 * Занести в лог отладочную информацию
 39 	 * @param {String} msg текст сообщения
 40 	 * @note Выводиться только в режиме отладки
 41 	 */
 42 	Ldebug : function(msg) {
 43 		if (NApi.debug.isDebug()) {
 44 			console.log("[DEBUG] " + msg);
 45 		}
 46 	}
 47 };
 48 
 49 
 50 (function() {
 51 	NApi.debug.enableDebugging();
 52 })();
