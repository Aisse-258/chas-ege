  3 /** @namespace NLib
  4  * Утилиты
  5  */
  6 var NLib = {
  7 	/** @namespace NLib._
  8 	 * Функционал, используемый только внутри модуля NLib
  9 	 * @private
 10 	 */
 11 	_ : {
 12 		/** @function NLib._.loadLibModule
 13 		 * Загрузить модуль NLib
 14 		 * @param {String} name название модуля
 15 		 * @private
 16 		 */
 17 		loadLibModule : function(name) {
 18 			document.write("<script charset=\"utf-8\" src=\"../lib/nlib/" + name + ".js\" onload=\"console.log('[NLib] Загружен модуль " + name + "');\"></script>");
 19 		}
 20 	},
 21 
 22 
 23 	/** @function NLib.getTypeOf
 24 	 * @param object объект
 25 	 * @return тип объекта ввиде [object ТИП]
 26 	 */
 27 	getTypeOf : function(object) {
 28 		return Object.prototype.toString.call(object);
 29 	},
 30 
 31 
 32 	/** @function NLib.toStringsArray
 33 	 * @param arr объект, которые необходимо превратить
 34 	 * @return arr ввиде массива строк
 35 	 */
 36 	toStringsArray : function(arr) {
 37 		switch (NLib.getTypeOf(arr)) {
 38 		case "[object Array]":
 39 			var newArr = [];
 40 			for (var i = 0; i < arr.length; i++) {
 41 				switch (NLib.getTypeOf(arr[i])) {
 42 				case "[object Number]":
 43 					newArr.push(arr[i].ts().toString());
 44 					break;
 45 				case "[object String]":
 46 					newArr.push(arr[i]);
 47 					break;
 48 				default:
 49 					throw TypeError("Параметр arr должен содержать только строки и числа");
 50 				}
 51 			}
 52 			return newArr;
 53 		case "[object Number]":
 54 			return [arr.ts()];
 55 		case "[object String]":
 56 			return [arr];
 57 		default:
 58 			throw TypeError("Параметр answers должен");
 59 		}
 60 	}
 61 };
