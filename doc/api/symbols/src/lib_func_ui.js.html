  4 function escapeFromIframe(){

 10 function getDocHeight(){
 11     var D = document;
 12     return Math.max(
 13         Math.max(D.body.scrollHeight, D.documentElement.scrollHeight),
 14         Math.max(D.body.offsetHeight, D.documentElement.offsetHeight),
 15         Math.max(D.body.clientHeight, D.documentElement.clientHeight)
 16     );
 17 }//Вроде как отсюда: http://james.padolsey.com/javascript/get-document-height-cross-browser/

 19 function catchTab(elem,key){
 20 	if(key.keyCode==9){
 21 		var n=elem.scrollTop;
 22 		var val=elem.value;
 23 		var sel=elem.selectionStart;
 24 		var rep=val.substr(elem.selectionStart-1,elem.selectionEnd-elem.selectionStart);
 25 		if(rep.match(/[\n\r]/)){
 26 			console.log(rep);
 27 			rep=rep.replace(/[\n](?![\n\r])/g,'\n\t');
 28 			rep=rep.replace(/[\r](?![\n\r])/g,'\r\t');
 29 			elem.value=val.substr(0,sel-1)+rep+val.substr(elem.selectionEnd-1);
 30 		}else{
 31 			elem.value=val.substr(0,sel)+'\t'+val.substr(elem.selectionEnd);
 32 			elem.selectionStart=elem.selectionEnd=sel+1;
 33 		}
 34 		elem.scrollTop=n;
 35 		return false;
 36 	}
 37 }

 39 function linkSpan(link,blank)
 40 {
 41 	if(blank)
 42 		window.open(link.replace("_","http://"));
 43 	else
 44 		self.location.replace(link.replace("_","http://"));
 45 }
