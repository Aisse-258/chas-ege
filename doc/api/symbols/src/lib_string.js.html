133 String.prototype.istDataToStd=function(){
134 /**Приводит дату, записанную в одной из общепринятых форм, к записи "по стандарту". Применяется только в комплексе заданий по истории.*/

173 String.prototype.plusminus=function(){
174 /**Примитивное упрощение математических выражений. Меняет "++" на "+", например.*/
175 	var a=this;
176 	for(;a.match(/[+-][+-]/);){
177 		a=a.replace(/[+][+]/g,'+');
178 		a=a.replace(/--/g,'+');
179 		a=a.replace(/[+]-/g,'-');
180 		a=a.replace(/-[+]/g,'-');
181 		a=a.replace(/[+]$/g,'');
182 		a=a.replace(/[{][+]/g,'{');
183 		a=a.replace(/[+][}]/g,'}');
184 		a=a.replace(/\(\+/g,'(');
185 		a=a.replace(/\+\)/g,')');
186 	}
187 	a=a.replace(/[=]\s*[+]/g,'=');
188 	a=a.replace(/[+]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'+');
189 	a=a.replace(/[-]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'-');
190 	a=a.replace(/[{]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'{');
191 	a=a.replace(/[}]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'}');
192 	a=a.replace(/[ ]1(?=[A-Za-zА-Яа-яЁё\\(])/g,' ');
193 	a=a.replace(/[~]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'~');
194 	a=a.replace(/[(]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'(');
195 	a=a.replace(/[)]1(?=[A-Za-zА-Яа-яЁё\\(])/g,')');
196 	a=a.replace(/[=]1(?=[A-Za-zА-Яа-яЁё\\(])/g,'=');
197 	a=a.replace(/[;]1(?=[A-Za-zА-Яа-яЁё\\(])/g,';');
198 	a=a.replace(/\^1(?=[A-Za-zА-Яа-яЁё\\(])/g,'^');
199 	a=a.replace(/\$1(?=[A-Za-zА-Яа-яЁё\\(])/g,'$');
200 	a=a.replace(/^1(?=[A-Za-zА-Яа-яЁё])/g,'');
201 	a=a.replace(/^[+]/g,'');
202 	a=a.replace(/[;][-]0/g,';0');
203 	a=a.reverse();
204 	a=a.replace(/[.]{2}(?=[A-Za-zА-Яа-яЁё])/g,'.');
205 	a=a.replace(/[.]{1}[$][.]{1}(?=[A-Za-zА-Яа-яЁё\\])/g,'$.');
206 	a=a.reverse();
207 	return a;
208 };
