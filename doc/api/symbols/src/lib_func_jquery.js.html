  4 function allCanvasToBackgroundImage(){
  5 	$('canvas').each(function(){
  6 		if(!this.style.backgroundImage)
  7 			this.style.backgroundImage='url('+this.toDataURL()+')';
  8 //		this.width=this.width;
  9 	});
 10 }

 12 function spoiler(){
 13 	$('.spoiler-body').hide();
 14 	$('.spoiler-hide').hide();
 15 	$('.spoiler-show').click(function(){
 16 		$(this).next().toggle();
 17 		$(this).next().next().slideToggle();
 18 		$(this).hide();
 19 	});
 20 	$('.spoiler-hide').click(function(){
 21 		$(this).next().slideToggle();
 22 		$(this).prev().toggle();
 23 		$(this).hide();
 24 	});
 25 }

 27 function allLinksToBlankTarget(){
 28 	$('a').each(function(){

 31 	});
 32 }

 34 function restoreLinksTarget(){

 40 function innerHTMLtoImg(elem){
 41 	html2canvas(elem, {
 42 		onrendered: function(canvas) {
 43 			var img=document.createElement('img');
 44 			img.src=canvas.toDataURL();
 45 			img.width=canvas.width;
 46 			img.height=canvas.height;
 47 			elem.innerHTML='';
 48 			elem.appendChild(img);
 49 		}
 50 	});
 51 }
 52 
 53 function replaceWithImg(elem){
 54 	html2canvas(elem, {
 55 		onrendered: function(canvas) {
 56 			var img=document.createElement('img');
 57 			img.src=canvas.toDataURL();
 58 			img.width=canvas.width;
 59 			img.height=canvas.height;
 60 			$(elem).replaceWith(img);
 61 		}
 62 	});
 63 }

 65 function allLinksToSpans(){
 66 	$('a').each(function(){
 67 		if(this.target=="_blank"){
 68 			$(this).replaceWith('<span class="spanlink" onclick="linkSpan(\''+
 69 				this.href.replace(/\/\/|https*:\/\//,'_')+'\',1);">'+this.innerHTML+'</span>');
 70 		}else{
 71 			$(this).replaceWith('<span class="spanlink" onclick="linkSpan(\''+
 72 				this.href.replace(/\/\/|https*:\/\//,'_')+'\');">'+this.innerHTML+'</span>');
 73 		}
 74 	});
 75 }
