  3 /** @function Array#getRandomItems
  4  * Получение случайные (не повторяющиеся) элементы массива
  5  * @param {Number} count кол-во случайных элеметнов
  6  * @return Массив случайных элементов массива
  7  */
  8 Array.prototype.getRandomItems = function(count) {
  9 	var items = [];
 10 	for (var i = 0; i < count; i++) {
 11 		if (this.length >= 1) {
 12 			items.push(this[NLib.math.randomize(0, this.length - 1)]);
 13 		} else {
 14 			items.push(this[0]);
 15 		}
 16 	}
 17 	return items;
 18 };

 22  * Получение случайный элемент массива
 23  * @return Случайный элемент массива
 24  */
 25 Array.prototype.getRandomItem = function() {
 26 	return this.getRandomItems(1);
