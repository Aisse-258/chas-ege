1 /** @function Array#get_random_items
  2  * Получение случайные (не повторяющиеся) элементы массива
  3  * @param {Number} count кол-во случайных элеметнов
  4  * @return Массив случайных элементов массива
  5  */
  6 Array.prototype.get_random_items = function(count) {
  7 	var items = [];
  8 	for (var i = 0; i < count; i++) {
  9 		if (this.length >= 1) {
 10 			items.push(this[NLib.math.randomize(0, this.length - 1)]);
 11 		} else {
 12 			items.push(this[0]);
 13 		}
 14 	}
 15 	return items;
 16 };

 21  * @return Случайный элемент массива
 22  */
 23 Array.prototype.get_random_item = function() {
 24 	return this.get_random_items(1);
