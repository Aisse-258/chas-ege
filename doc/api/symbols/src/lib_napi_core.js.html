1 "use strict";
  2 
  3 /** @namespace NApi
  4  * New API [на́пи]
  5  */
  6 var NApi = {
  7 	/** @namespace NApi.info
  8 	 * Информация об API
  9 	 */
 10 	info : {
 11 		/**
 12 		 * Версия API
 13 		 */
 14 		API_VERSION : { major : 0, minor : 0 },
 15 
 16 
 17 		/**
 18 		 * Выпуск приложения
 19 		 */
 20 		APPLICATION_RELEASE : "__APPLICATION_RELEASE____",
 21 
 22 
 23 		/**
 24 		 * Версия в которой была (последний раз) сломана обратная совметсимость
 25 		 */
 26 		LAST_BACKWARD_COMPATIBILITY_BREAK : { major : 0, minor : 0 },
 27 
 28 
 29 		/** @function NApi.info.requireApiVersion
 30 		 * Проверить совместимость с текущего API, с API версии major.minor
 31 		 * @param {Number} major
 32 		 * @param {Number} minor
 33 		 */
 34 		requireApiVersion : function(major, minor) {
 35 			if (major < NApi.info.LAST_BACKWARD_COMPATIBILITY_BREAK.major || minor < NApi.info.LAST_BACKWARD_COMPATIBILITY_BREAK.minor) {
 36 				throw new Error("Была запрошена устаревшая версия NApi");
 37 			} else if (major > NApi.info.API_VERSION.major || minor > NApi.info.API_VERSION.minor) {
 38 				throw new Error("Была запрошена более новая версия NApi");
 39 			}
 40 		}
 41 	},
 42 
 43 
 44 	/** @namespace NApi._
 45 	 * Функционал, используемый только внутри самого NApi
 46 	 * @private
 47 	 */
 48 	_ : {
 49 		/** @function NApi._.Linfo
 50 		 * Занести в лог информацию
 51 		 * @param {String} msg текст сообщения
 52 		 * @private
 53 		 */
 54 		Linfo : function(msg) {
 55 			console.log("[NApi][info] " + msg);
 56 		},
 57 
 58 
 59 		/** @function NApi._.Lmsg
 60 		 * Занести в лог сообщение
 61 		 * @param {String} msg текст сообщения
 62 		 * @private
 63 		 */
 64 		Lmsg : function(msg) {
 65 			console.log("[NApi][msg] " + msg);
 66 		},
 67 
 68 
 69 		/** @function NApi._.Lwarn
 70 		 * Занести в лог предупреждение
 71 		 * @param {String} msg текст сообщения
 72 		 * @private
 73 		 */
 74 		Lwarn : function(msg) {
 75 			console.log("[NApi][WARN] " + msg);
 76 		},
 77 
 78 
 79 		/** @function NApi._.Lerr
 80 		 * Занести в лог информацию об ощибке
 81 		 * @param {String} msg текст сообщения
 82 		 * @private
 83 		 */
 84 		Lerr : function(msg) {
 85 			console.log("[NApi][ERROR] " + msg);
 86 		},
 87 
 88 
 89 		/** @function NApi._.loadApiModule
 90 		 * Загрузить модуль API
 91 		 * @param {String} name название модуля
 92 		 * @private
 93 		 */
 94 		loadApiModule : function(name) {
 95 			document.write("<script charset=\"utf-8\" src=\"../lib/napi/" + name + ".js\" onload=\"NApi._.Linfo('Загружен модуль " + name + "');\"></script>");
 96 		}
 97 	},
 98 
 99 
100 	/** @function NApi.Linfo
101 	 * Занести в лог информацию
102 	 * @param {String} msg текст сообщения
103 	 */
104 	Linfo : function(msg) {
105 		console.log("[info] " + msg);
106 	},
107 
108 
109 	/** @function NApi.Lmsg
110 	 * Занести в лог сообщение
111 	 * @param {String} msg текст сообщения
112 	 */
113 	Lmsg : function(msg) {
114 		console.log("[msg] " + msg);
115 	},
116 
117 
118 	/** @function NApi.Lwarn
119 	 * Занести в лог предупреждение
120 	 * @param {String} msg текст сообщения
121 	 */
122 	Lwarn : function(msg) {
123 		console.log("[WARN] " + msg);
124 	},
125 
126 
127 	/** @function NApi.Lerr
128 	 * Занести в лог информацию об ощибке
129 	 * @param {String} msg текст сообщения
130 	 */
131 	Lerr : function(msg) {
132 		console.log("[ERROR] " + msg);
133 	},
134 };
135 
136

138 	NApi._.Linfo("NApi v" + NApi.info.API_VERSION.major + "." + NApi.info.API_VERSION.minor);
139 
140 	NApi._.Linfo("Загрузка модулей...");
141 	NApi._.loadApiModule("debug");
142 	NApi._.loadApiModule("task");
143 	NApi._.loadApiModule("ui");
144
