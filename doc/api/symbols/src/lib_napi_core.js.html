1 /** @namespace NApi
  2  * New API [на́пи]
  3  */
  4 var NApi = {
  5 	/** @namespace NApi.info
  6 	 * Информация об API
  7 	 */
  8 	info : {
  9 		/**
 10 		 * Версия API
 11 		 */
 12 		VERSION : "InDev"
 13 	},
 14 
 15 
 16 	/** @namespace NApi._
 17 	 * Функционал, используемый только внутри самого NApi
 18 	 * @private
 19 	 */
 20 	_ : {
 21 		/**
 22 		 * Флаг отладочного режима
 23 		 */
 24 		debug_mode : false,
 25 
 26 
 27 		/** @function NApi._.Linfo
 28 		 * Занести в лог информацию
 29 		 * @param {String} msg текст сообщения
 30 		 * @private
 31 		 */
 32 		Linfo : function(msg) {
 33 			console.log("[NApi][info] " + msg);
 34 		},
 35 
 36 
 37 		/** @function NApi._.Lmsg
 38 		 * Занести в лог сообщение
 39 		 * @param {String} msg текст сообщения
 40 		 * @private
 41 		 */
 42 		Lmsg : function(msg) {
 43 			console.log("[NApi][msg] " + msg);
 44 		},
 45 
 46 
 47 		/** @function NApi._.Lwarn
 48 		 * Занести в лог предупреждение
 49 		 * @param {String} msg текст сообщения
 50 		 * @private
 51 		 */
 52 		Lwarn : function(msg) {
 53 			console.log("[NApi][WARN] " + msg);
 54 		},
 55 
 56 
 57 		/** @function NApi._.Lerr
 58 		 * Занести в лог информацию об ощибке
 59 		 * @param {String} msg текст сообщения
 60 		 * @private
 61 		 */
 62 		Lerr : function(msg) {
 63 			console.log("[NApi][ERROR] " + msg);
 64 		},
 65 
 66 
 67 		/** @function NApi._.Ldebug
 68 		 * Занести в лог отладочную информацию
 69 		 * @param {String} msg текст сообщения
 70 		 * @note Выводиться только в режиме отладки
 71 		 * @private
 72 		 */
 73 		Ldebug : function(msg) {
 74 			if (NApi.state.is_debug()) {
 75 				console.log("[NApi][DEBUG] " + msg);
 76 			}
 77 		},
 78 
 79 
 80 		/** @function NApi._.load_js
 81 		 * Загрузить .js-файл
 82 		 * @param {String} path путь к .js-файлу
 83 		 * @private
 84 		 */
 85 		load_js : function(path) {
 86 			document.write("<script charset=\"utf-8\" src=\"" + path + "\" onload=\"console.log('Loaded " + path + "');\"></script>");
 87 		},
 88 
 89 
 90 		/** @function NApi._.load_api_module
 91 		 * Загрузить модуль API
 92 		 * @param {String} name название модуля
 93 		 * @private
 94 		 */
 95 		load_api_module : function(name) {
 96 			document.write("<script charset=\"utf-8\" src=\"../lib/napi/" + name + ".js\" onload=\"NApi._.Linfo('Loaded API module " + name + "');\"></script>");
 97 		}
 98 	},
 99 
100 
101 	/** @namespace NApi.state
102 	 * Состояние приложения
103 	 */
104 	state : {

106 		 * Занести в лог информацию
107 		 * @return {Boolean} включен ли режим отладки
108 		 */
109 		is_debug : function() {
110 			return NApi._.debug_mode;
111 		},
112 
113 
114 		/** @function NApi._.enable_debug_mode

116 		 */
117 		enable_debug_mode : function() {
118 			NApi._.debug_mode = true;
119 			NApi._.Linfo("debug-mode enabled");
120 		},
121 
122 
123 		/** @function NApi._.disable_debug_mode
124 		 * Выключить отладочный режим
125 		 */
126 		disable_debug_mode : function() {

128 			NApi._.Linfo("debug-mode disabled");
129 		},
130 	},
131 
132 
133 	/** @function NApi.Linfo
134 	 * Занести в лог информацию
135 	 * @param {String} msg текст сообщения

138 		console.log("[info] " + msg);
139 	},
140 
141 
142 	/** @function NApi.Lmsg
143 	 * Занести в лог сообщение
144 	 * @param {String} msg текст сообщения
145 	 */
146 	Lmsg : function(msg) {
147 		console.log("[msg] " + msg);
148 	},
149 
150 
151 	/** @function NApi.Lwarn
152 	 * Занести в лог предупреждение
153 	 * @param {String} msg текст сообщения
154 	 */
155 	Lwarn : function(msg) {
156 		console.log("[WARN] " + msg);
157 	},
158 
159

161 	 * Занести в лог информацию об ощибке
162 	 * @param {String} msg текст сообщения

171 	 * @param {String} msg текст сообщения
172 	 * @note Выводиться только в режиме отладки
173 	 */
174 	Ldebug : function(msg) {
175 		if (NApi.state.is_debug()) {
176 			console.log("[DEBUG] " + msg);
