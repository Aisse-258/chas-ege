1 'use strict';
  2 /*
  3 Атрибуты (HTML5-data-)
  4 * data-jstorage-id - идентификатор элемента в jStorage. Обязателен!
  5 * data-jstorage-ne - не отслеживать (через пробел)
  6 * 	zn - значение
  7 * 	vi - видимость
  8 * 	ih - innerHTML
  9 */
 10 
 11 /*
 12 Attributes (HTML5-data-)
 13 * data-jstorage-id - identifier to link the element in jStorage with. Necessary!
 14 * data-jstorage-ne - do not track (divided by space)
 15 * 	zn - value
 16 * 	vi - visibility
 17 * 	ih - innerHTML
 18 */
 19 
 20 //Загрузить данные
 21 //Load data
 22 $.jStorage.zagrData=function(){
 23 	var storedData=$.jStorage.get('data-jstorage');
 24 	if(!storedData)
 25 		return;
 26 	[].slice.call(document.querySelectorAll('*[data-jstorage-id]'),0).map(
 27 		function(th){'use strict';
 28 			try{
 29 				var stor=storedData[th.getJStorageId()];
 30 				if(stor){
 31 					var ne=th.getAttribute('data-jstorage-ne');
 32 					ne=ne?ne:'';
 33 					if(!ne.match('zn')){
 34 						th.checked=stor.checked;
 35 						th.value=stor.value;
 36 					}
 37 					if(!ne.match('ih'))
 38 						th.innerHTML=stor.innerHTML;
 39 					if(!ne.match('vi'))
 40 						th.style.display=stor.style.display;
 41 				}
 42 			}catch(e){
 43 				console.log(th,' - ошибка в jStorage.zapomni');
 44 				console.log(th,' - error in jStorage.zapomni');
 45 			}
 46 		}
 47 	);
 48 };
 49 
 50 //Сохранить данные
 51 //Save data
 52 $.jStorage.sohrData=function(){
 53 	var storedData=$.jStorage.get('data-jstorage');
 54 	if(!storedData)
 55 		storedData={};
 56 	[].slice.call(document.querySelectorAll('*[data-jstorage-id]'),0).map(
 57 		function(th){'use strict';
 58 			try{
 59 				var jstid=th.getJStorageId();
 60 				var stor=storedData[jstid];
 61 				if(!stor){
 62 					storedData[jstid]={};
 63 					stor=storedData[jstid];
 64 				}
 65 				var ne=th.getAttribute('data-jstorage-ne');
 66 				ne=ne?ne:'';
 67 				if(!ne.match('zn')){
 68 					stor.checked=th.checked;
 69 					stor.value=th.value;
 70 				}
 71 				if(!ne.match('ih'))
 72 					stor.innerHTML=th.innerHTML;
 73 				if(!stor.style)
 74 					stor.style={};
 75 				if(!ne.match('vi'))
 76 					stor.style.display=th.style.display;
 77 			}catch(e){
 78 				console.log(th,' - ошибка в jStorage.zapomni');
 79 				console.log(th,' - error in jStorage.zapomni');
 80 			}
 81 		}
 82 	);
 83 	$.jStorage.set('data-jstorage',storedData)
 84 };
 85 
 86 HTMLElement.prototype.getJStorageId=function(){
 87 	return this.getAttribute('data-jstorage-id');
 88 }
 89 
 90 //Объявили свой метод у HTMLElement.prototype - сделаем его неперечислимым
 91 //Since we have created a new method in HTMLElement.prototype , we should make the method unenumerable
 92 Object.defineProperty(HTMLElement.prototype, "getJStorageId", { enumerable: false });
